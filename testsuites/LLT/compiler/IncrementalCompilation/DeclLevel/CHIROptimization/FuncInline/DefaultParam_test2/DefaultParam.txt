func foo<T>(a!: Int32 = 1): Int32 {
    if (a == 0) {
        return 0
    } else if (a == 1) {
        return foo<Int64>(a: 2)
    } else {
        return 2
    }
}

main(): Int32 {
    foo<Int8>()
}

foo(l:8, c:1) -> main(l:18, c:1),
